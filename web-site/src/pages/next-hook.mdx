## NextHook

NextHooks are hooks to update the state.

## Basic Usage

```code lines=6,9,13
const store = createStore({
  state: { name: 'John Doe' }
});

const useAppState = createStateHook(AppStoreProvider);
const useNextAppState = createNextHook(AppStoreProvider);

const Hello:React.FC = () => {
  const next = useNextAppState(state => state);
  const name = useAppState(state => state.name);

  function setName(e) {
    next(state => { state.name = e.target.value});
  }

  return <input value={name} onChange={setName}/>
}
```

If you want to simplify and unify the access to portion of your states you utilize a **scoped next state hook**:

```code lines=3-5,12,19
const store = createStore({
  state: {
    user: {
      name: 'John Doe'
    }
  }
});

...

// scoped next hook
const useNextUserState  = createNextHook(AppStoreProvider, state => state.user);

const Hello = () => {
  const nextUserState = useNextUserState(state => state);

  function setName(e) {
    // nextUserState provides the `user` object
    nextUserState(user => { user.name = e.target.value});
  }
  ...
}
```
